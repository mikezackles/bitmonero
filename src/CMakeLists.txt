include(cotire)

add_definitions(-DSTATICLIB)

file(GLOB_RECURSE COMMON common/*)
file(GLOB_RECURSE CRYPTO crypto/*)
file(GLOB_RECURSE CRYPTONOTE_CORE cryptonote_core/*)
file(GLOB_RECURSE CRYPTONOTE_PROTOCOL cryptonote_protocol/*)
file(GLOB_RECURSE DAEMON daemon/*.cpp)
file(GLOB_RECURSE DAEMONIZER daemonizer/*.cpp)
if(WIN32)
  file(GLOB_RECURSE daemon_posix_files daemonizer/posix_*.cpp)
  foreach(posix_file ${daemon_posix_files})
    list(REMOVE_ITEM DAEMONIZER "${posix_file}")
  endforeach()
else()
  file(GLOB_RECURSE daemon_windows_files daemonizer/windows_*.cpp)
  foreach(windows_file ${daemon_windows_files})
    list(REMOVE_ITEM DAEMONIZER "${windows_file}")
  endforeach()
endif()
file(GLOB_RECURSE P2P p2p/*)
file(GLOB_RECURSE EPEE epee/*)
file(GLOB_RECURSE RPC rpc/*)
file(GLOB_RECURSE SIMPLEWALLET simplewallet/*)
file(GLOB_RECURSE CONN_TOOL connectivity_tool/*)
file(GLOB_RECURSE WALLET wallet/*)
file(GLOB_RECURSE RPCWALLET rpcwallet/*)
file(GLOB_RECURSE MINER miner/*)

source_group(common FILES ${COMMON})
source_group(crypto FILES ${CRYPTO})
source_group(cryptonote_core FILES ${CRYPTONOTE_CORE})
source_group(cryptonote_protocol FILES ${CRYPTONOTE_PROTOCOL})
source_group(daemon FILES ${DAEMON})
source_group(p2p FILES ${P2P})
source_group(epee FILES ${EPEE})
source_group(rpc FILES ${RPC})

if (BUILD_SIMPLE_WALLET)
  source_group(simplewallet FILES ${SIMPLEWALLET})
endif ()
if (BUILD_CONN_TOOL)
  source_group(connectivity-tool FILES ${CONN_TOOL})
endif ()
source_group(wallet FILES ${WALLET})
source_group(rpcwallet FILES ${RPCWALLET})

if (BUILD_SIMPLE_MINER)
  source_group(simpleminer FILES ${MINER})
endif ()

add_library(common ${COMMON})
add_library(crypto ${CRYPTO})
add_library(cryptonote_core ${CRYPTONOTE_CORE})
add_library(daemonizer STATIC ${DAEMONIZER})

add_executable(daemon ${DAEMON})

if (BUILD_RPC_WALLET)
  add_executable(rpcwallet ${RPCWALLET} )
  target_link_libraries(rpcwallet daemonizer wallet rpc cryptonote_core crypto common upnpc-static ${Boost_LIBRARIES} ${CMAKE_THREAD_LIBS_INIT})
  set(OPTIONAL ${OPTIONAL} rpcwallet)
  if (ENABLE_COTIRE)
    cotire(rpcwallet)
  endif()
endif()

if (BUILD_CONN_TOOL)
  add_executable(connectivity_tool ${CONN_TOOL})
  target_link_libraries(connectivity_tool cryptonote_core crypto common ${CMAKE_THREAD_LIBS_INIT} ${Boost_LIBRARIES})
  set(OPTIONAL ${OPTIONAL} connectivity_tool)
  if (ENABLE_COTIRE)
    cotire(connectivity_tool)
  endif ()
endif ()

if (BUILD_SIMPLE_MINER)
  add_executable(simpleminer ${MINER})
  target_link_libraries(simpleminer cryptonote_core crypto common ${CMAKE_THREAD_LIBS_INIT} ${Boost_LIBRARIES})
  set(OPTIONAL ${OPTIONAL} simpleminer)
  if (ENABLE_COTIRE)
    cotire(simpleminer)
  endif ()
endif()

target_link_libraries(daemon daemonizer rpc cryptonote_core crypto common upnpc-static otshell_utils ${CMAKE_THREAD_LIBS_INIT} ${Boost_LIBRARIES} )
# ${LEVELDB_LIBS}) - TODO remove me later, I'm example

if (ENABLE_COTIRE)
  cotire(daemon)
endif ()

add_library(rpc ${RPC})
add_library(wallet ${WALLET})

if (BUILD_SIMPLE_WALLET)
  add_executable(simplewallet ${SIMPLEWALLET})
  target_link_libraries(simplewallet wallet rpc cryptonote_core crypto common upnpc-static ${CMAKE_THREAD_LIBS_INIT} ${Boost_LIBRARIES})
  add_dependencies(simplewallet version)
  set(OPTIONAL ${OPTIONAL} simplewallet)
  if (ENABLE_COTIRE)
    cotire(simplewallet)
  endif ()
endif ()

add_dependencies(daemon version)
add_dependencies(rpc version)

set_property(TARGET common crypto cryptonote_core rpc wallet PROPERTY FOLDER "libs")
set_property(TARGET daemon ${OPTIONAL} PROPERTY FOLDER "prog")
if (STATIC)
  set_property(TARGET daemon ${OPTIONAL} PROPERTY LINK_SEARCH_START_STATIC 1)
  set_property(TARGET daemon ${OPTIONAL} PROPERTY LINK_SEARCH_END_STATIC 1)
endif()
set_property(TARGET daemon PROPERTY OUTPUT_NAME "bitmonerod")
